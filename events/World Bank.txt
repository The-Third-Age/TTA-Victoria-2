			limit = {
				has_country_modifier = alliance_of_dwarves_and_men

				is_greater_power = no

					ai = yes # hitting every ai realm in case a new alliance member GP shows up. This won't cover a player, but it'll be rare that a not hitting a player-realm will actually affect gameplay

			limit = { has_country_modifier = alliance_of_dwarves_and_men }
