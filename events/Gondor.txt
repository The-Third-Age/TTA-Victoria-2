# The Death of Ecthelion II and Coronation of Denethor II {
# 	Currently it is assumed that Denethor II will succeed Ecthelion II as there is no way for him to die early or do other things
# The Death of Ecthelion II
country_event = {
	id = 50021
	title = "The Death of Ecthelion II"
	desc = "EVTDESC50021"
	picture = "gondor/leader_death"
	fire_only_once = yes
	
	trigger = {
		year = 2980
		nationalvalue = ecthelion_ii
		NOT = { is_substate = yes }
	}
	
	mean_time_to_happen = {
		months = 96
		
		modifier = {
			factor = 0.75
			year = 2982
			NOT = { year = 2983 }
		}
		
		modifier = {
			factor = 0.5
			year = 2983
			NOT = { year = 2984 }
		}
		
		modifier = {
			factor = 0.25
			year = 2984
		}
		
		modifier = {
			factor = 0.01
			year = 2985
		}
	}
	
	option = {
		name = "He will be missed"
		
		# If Denethor II is alive, we know the next leader
		random_owned = {
			limit = {
				owner = { has_global_flag = denethor_ii_alive }
			}
			owner = {
				set_country_flag = knows_next_leader
				
				any_country = {
					limit = {
						exists = yes
						nationalvalue = ecthelion_ii
					}
					set_country_flag = knows_next_leader
				}
			}
		}
		
		# Kills Ecthelion II and coronates the new leader
		country_event = 3501
		
		any_country = {
			limit = {
				exists = yes
				nationalvalue = ecthelion_ii
			}
			country_event = 3501
		}
		
		random_owned = {
			limit = {
				owner = { has_global_flag = denethor_ii_alive }
			}
			owner = {
				nationalvalue = denethor_ii
				
				any_country = {
					limit = {
						exists = yes
						nationalvalue = ecthelion_ii
					}
					nationalvalue = denethor_ii
				}
			}
		}
		
		clr_global_flag = ecthelion_ii_alive
		
		# Invites non-puppet allies to the coronation
		any_country = {
			limit = {
				alliance_with = THIS
				NOT = { is_substate = yes }
			}
			country_event = { id = 3507 }
		}
		
		# Denethor hated and was jealous of Thorongil. If Thorongil was still an advisor when Ecthelion II died, he would have left/been expelled from the court
		remove_country_modifier = thorongil
	}
}

# The Death of Denethor II
country_event = {
	id = 50049
	title = "The Passing of Denethor II"
	desc = "EVTDESC50049"
	picture = "gondor/leader_death"
	
	fire_only_once = yes
	
	trigger = {
		year = 3025
		nationalvalue = denethor_ii
		NOT = { is_substate = yes }
	}
	
	mean_time_to_happen = {
		months = 96
		
		modifier = {
			factor = 0.75
			year = 3027
			NOT = { year = 3028 }
		}
		
		modifier = {
			factor = 0.5
			year = 3028
			NOT = { year = 3029 }
		}
		
		modifier = {
			factor = 0.25
			year = 3029
		}
		
		modifier = {
			factor = 0.01
			year = 3030
		}
	}
	
	option = {
		name = "He will be missed"
		
		# If one of Denethor II's successors is alive, we know the next leader
		random_owned = {
			limit = {
				owner = {
					OR = {
						has_global_flag = boromir_ii_alive
						has_global_flag = faramir_i_alive
						# TODO: global flag about Aragorn coming to Gondor (being available to be King)
					}
				}
			}
			owner = { 
				set_country_flag = knows_next_leader
				
				any_country = {
					limit = {
						exists = yes
						nationalvalue = denethor_ii
					}
					set_country_flag = knows_next_leader
				}
			}
		}
		
		# Kills Denethor II and coronates the new leader
		country_event = 3501
		
		any_country = {
			limit = {
				exists = yes
				nationalvalue = denethor_ii
			}
			country_event = 3501
		}
		
		# Boromir II highest priority
		random_owned = {
			limit = {
				owner = { has_global_flag = boromir_ii_alive }
			}
			owner = {
				nationalvalue = boromir_ii
				
				any_country = {
					limit = {
						exists = yes
						nationalvalue = denethor_ii
					}
					nationalvalue = boromir_ii
				}
			}
		}
		
		# Faramir I next
		random_owned = {
			limit = {
				owner = {
					NOT = { has_global_flag = boromir_ii_alive }
					has_global_flag = faramir_i_alive
				}
			}
			owner = {
				nationalvalue = faramir_i
				
				any_country = {
					limit = {
						exists = yes
						nationalvalue = denethor_ii
					}
					nationalvalue = faramir_i
				}
			}
		}
		
		# Aragorn Elessar last
		# TODO: Setup the check for Aragorn
		
		clr_country_flag = denethor_ii_alive
		
		# Invites non-puppet allies to the coronation
		any_country = {
			limit = {
				alliance_with = THIS
				NOT = { is_substate = yes }
			}
			country_event = { id = 3507 }
		}
	}
}
