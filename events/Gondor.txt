# The Tolfalas Question

	id = 50000
	picture = tolfalas_map
	title = "The Tolfalas Question"
	desc = "EVTDESC50000"
	
	is_triggered_only = yes
	
	option = {
		name = "§BTo Dol Amroth§!"
		AMR = {
			inherit = TOL
			any_owned = {
				limit = {
					is_core = TOL
				}
				remove_core = TOL
				add_core = AMR
			}
			prestige = 5
		}
		relation = {
			who = AMR
			value = 50
		}
		
		ai_chance = {
			factor = 90
		}
	}
	
	option = {
		name = "For us"
		inherit = TOL
		any_owned = {
			limit = {
				is_core = TOL
			}
			remove_core = TOL
			add_core = AMR
		}
		relation = {
			who = AMR
			value = -200
		}
		set_global_flag = no_tolfalas_for_da
		
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				has_global_flag = railroading_enabled
			}
		}
	}
}

# Dol Amroth wins
country_event = {
	id = 50002
	picture = dol_amroth
	title = "We are Victorious!"
	desc = "EVTDESC50002"
	
	fire_only_once = yes
	
	trigger = {
		has_global_flag = da_independence_war
		tag = AMR
		NOT = { war_with = GON }
		NOT = { vassal_of = GON }
		truce_with = GON
	}
	
	option = {
		name = "Freedom!"
		GON_56 = { secede_province = AMR }
		prestige = 25
		all_core = {
			remove_core = GON
		}
		GON = { prestige = -50 }
	}
}
