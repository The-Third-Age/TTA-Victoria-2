# Harondor colonisation by the Haradrim/Corsairs
country_event = {
	id = 60200
	title = "The colonisation of Harondor"
	desc = "EVTDESC60200"
	picture = "harad/map"
	
	trigger = {
		has_country_flag = colonizing_harondor
		NOT = { has_country_flag = colonized_harondor }
		any_owned_province = {
			has_empty_adjacent_province = yes
			is_core = HRD
		}
	}
	
	mean_time_to_happen = {
		months = 6
	}
	
	option = {
		name = "We are expanding"
		random_owned = {
			limit = {
				has_empty_adjacent_province = yes
				is_core = HRD
				owner = { tag = THIS }
			}
			random_empty_neighbor_province = {
				limit = { is_core = HRD }
				random_owned = {
					limit = { THIS = { NOT = { slavery = no_slavery } } }
					state_scope = { is_slave = yes }
				}
				secede_province = THIS
			}
		}
	}
}
